noelle char lvl=90/90 cons=6 talent=9,9,9;
noelle add weapon="serpentspine" refine=1 lvl=90/90  +params=[stacks=5];
noelle add set="husk" count=4;            // Can't run +params=[stacks=4] with SS
#noelle add set="archaicpetra" count=4;   // Roughly same results
noelle add stats hp=4780 atk=311 def%=0.583 geo%=0.466 cd=0.622;
noelle add stats hp=507.88 hp%=0.0991 atk=33.07 atk%=0.0991 def=39.35 def%=0.062*2 em=39.63 er=0.0551*6 cr=0.0331*10 cd=0.0662*10;

nicole char lvl=90/90 cons=0 talent=9,9,9; 
nicole add weapon="flowingpurity" refine=1 lvl=90/90;
nicole add set="hg" count=4;
nicole add stats hp=4780 atk=311 atk%=0.466 atk%=0.466 atk%=0.466;
nicole add stats def%=0.062*2 def=19.68*2 hp=253.94*2 hp%=0.0496*2 atk=16.54*10 atk%=0.0496*6 er=0.0551*10 em=19.82*2 cr=0.0331*2 cd=0.0662*2;

xilonen char lvl=90/90 cons=0 talent=9,9,9; 
xilonen add weapon="favsword" refine=3 lvl=90/90;
xilonen add set="scroll" count=4;
xilonen add stats hp=4780 atk=311 def%=0.583 def%=0.583 cr=0.311; #main
xilonen add stats def%=0.248 def=39.36 hp=507.88 hp%=0.0992 atk=33.08 atk%=0.0992 er=0.1102 em=39.64 cr=0.331 cd=0.7944;

# Options
options iteration=1000; 
options swap_delay=12;
energy every interval=480,720 amount=1;

# Targets
target lvl=100 resist=0.1 radius=2 pos=0,2.3 hp=100000000;

# Rotation
active noelle; # SS Stacks

for let i = 0; i < 4; i = i + 1 {
  
  nicole skill;
  if .nicole.burst.ready {
    nicole burst;
  }
  
  durin attack, skill:2, burst, attack;   // Weaving apparently matters?
  xilonen skill, attack:2;
  // noelle swap; pick_up_crystallize("pyro");
  noelle burst, skill, attack:3, dash, attack:3, dash, attack:3;
  
  durin attack, skill:2;
  xilonen skill, attack:2;
  // noelle swap; pick_up_crystallize("pyro");
  noelle attack:4, dash, attack:4;
}
